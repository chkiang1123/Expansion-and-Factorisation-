Preparing battle..."); // Host dictates when the game starts for both players setTimeout(() => { sendNetworkData({ type: 'start_game' }); startPvP(); }, 1500); }); }); peer.on('error', (err) => { updatePvPStatus("Error: Code might be in use or server busy."); }); } window.setupJoin = function() { mode = 'pvp'; isHost = false; let displayId = document.getElementById('join-id').value.trim(); if(!displayId) { updatePvPStatus("Please enter a 3-Digit Code!"); return; } updatePvPStatus("Connecting..."); window.cleanupNetwork(); peer = new Peer(); peer.on('open', (id) => { conn = peer.connect('mathbattle-' + displayId); conn.on('open', () => { setupConnectionListeners(); updatePvPStatus("Connected! Waiting for host to start..."); // FIX: Removed the setTimeout here. Guest now waits for Host's signal. }); conn.on('error', (err) => { updatePvPStatus("Failed to connect to Host."); }); }); peer.on('error', (err) => { updatePvPStatus("Error connecting to server."); }); } function setupConnectionListeners() { conn.on('data', (data) => { handleNetworkData(data); }); conn.on('close', () => { alert("Opponent disconnected!"); location.reload(); }); } function sendNetworkData(payload) { if (conn && conn.open) { conn.send(payload); } } function handleNetworkData(data) { // FIX: New listener for the game start signal if(data.type === 'start_game') { startPvP(); } else if(data.type === 'question') { myStatus = 'pending'; opponentStatus = 'pending'; turnActive = true; renderQuestionUI(data.qData.q, data.qData.ans, data.options); resetTimer(); } else if(data.type === 'correct') { if(!turnActive) return; turnActive = false; opponentStatus = 'correct'; clearInterval(timerInterval); disableButtons(); animateSprite('enemy-sprite'); playerHP -= data.amount; showDamage('player-sprite', data.amount); document.getElementById('question-area').innerText = `Opponent answered first! You took ${data.amount} damage!`; setTimeout(() => { updateHP(); if(!checkWinLoss() && isHost) setTimeout(sendPvPQuestion, 1500); }, 1000); } else if(data.type === 'wrong') { opponentStatus = 'wrong'; if (myStatus === 'wrong' && turnActive) handleBothWrong(); } }
