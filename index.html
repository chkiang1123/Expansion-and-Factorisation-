function setupHost(level) { mode = 'pvp'; difficulty = level; isHost = true; updatePvPStatus("Creating Room..."); // Generates a 3-digit number for the student to see let displayId = Math.floor(100 + Math.random() * 900).toString(); // Adds a secret prefix so it never collides with other people on the internet let actualRoomId = "hkdse-math-" + displayId; peer = new Peer(actualRoomId, peerConfig); // Show the simple 3-digit ID to the student peer.on('open', (id) => updatePvPStatus(`Room Created! ID: ${displayId}\nTell your opponent to enter this!`)); peer.on('connection', (connection) => { conn = connection; startPvP(); }); peer.on('error', (err) => updatePvPStatus(`Error: ${err.type}. Try again.`)); } function setupJoin() { mode = 'pvp'; isHost = false; let userInput = document.getElementById('join-id').value.trim(); if(!userInput) { updatePvPStatus("Please enter a Room ID!"); return; } // Add the same secret prefix to find the exact room let joinId = "hkdse-math-" + userInput; updatePvPStatus("Connecting..."); peer = new Peer(undefined, peerConfig); peer.on('open', () => { conn = peer.connect(joinId); conn.on('error', (err) => updatePvPStatus(`Connection failed: ${err.message}`)); startPvP(); }); peer.on('error', (err) => updatePvPStatus(`Error: ${err.type}. Check Room ID.`)); }
